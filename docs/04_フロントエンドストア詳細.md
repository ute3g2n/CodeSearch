## 1. WorkspaceStore

**ファイル:** `src/stores/workspace.ts`

```typescript
interface WorkspaceState {
  // ----- State -----
  currentWorkspace: Workspace | null;
  recentWorkspaces: Workspace[];
  hasIndexWriteLock: boolean;
  indexStatus: IndexStatus;

  // ----- Actions -----

  /** フォルダ選択ダイアログを表示し、選択されたパスでワークスペースを開く */
  openWorkspaceDialog: () => Promise<void>;

  /** 指定パスのワークスペースを開く
   *  処理:
   *  1. invoke("open_workspace", { path })
   *  2. state更新（currentWorkspace, hasIndexWriteLock, indexStatus）
   *  3. EditorStore.openWelcomeTab() をクローズ
   *  4. EditorStore.restoreTabs() を呼び出し（前回のタブ復元）
   *  5. BookmarkStore.loadBookmarks() を呼び出し
   */
  openWorkspace: (path: string) => Promise<void>;

  /** 現在のワークスペースを閉じる */
  closeWorkspace: () => Promise<void>;

  /** 最近のワークスペース一覧を読み込む */
  loadRecentWorkspaces: () => Promise<void>;

  /** インデックス状態を更新する（イベントハンドラから呼ばれる） */
  updateIndexStatus: (status: Partial<IndexStatus>) => void;
}
```

**イベントリスナー（初期化時に登録）:**

| イベント | ハンドラ |
|---------|---------|
| `index://progress` | `NotificationStore.updateProgress()` |
| `index://ready` | `updateIndexStatus({ state: 'ready', ... })` + `NotificationStore.addNotification()` |
| `index://error` | `updateIndexStatus({ state: 'error', ... })` + `NotificationStore.addNotification()` |
| `index://updated` | (特になし、検索結果はユーザー操作で再取得) |

## 2. EditorStore

**ファイル:** `src/stores/editor.ts`

```typescript
interface EditorState {
  // ----- State -----
  groups: EditorGroup[];
  activeGroupId: string;
  /** ファイル内容キャッシュ（パス → FileContent） */
  fileContentCache: Map<string, FileContent>;

  // ----- Actions -----

  /** ファイルをタブとして開く
   *  - 既に開いている場合はそのタブをアクティブにする
   *  - lineNumber指定があればそこにスクロール
   */
  openFile: (path: string, options?: { lineNumber?: number; groupId?: string }) => Promise<void>;

  /** ウェルカムタブを開く（ワークスペース未選択時に自動呼び出し） */
  openWelcomeTab: () => void;

  /** 検索エディタタブを開く */
  openSearchEditor: (query?: string) => void;

  /** プレーンテキストタブを開く（検索結果のエディタ表示用） */
  openPlainText: (title: string, content: string) => void;

  /** タブを閉じる */
  closeTab: (groupId: string, tabId: string) => void;

  /** 指定タブ以外を閉じる */
  closeOtherTabs: (groupId: string, tabId: string) => void;

  /** 指定タブの右側をすべて閉じる */
  closeTabsToRight: (groupId: string, tabId: string) => void;

  /** グループ内の全タブを閉じる */
  closeAllTabs: (groupId: string) => void;

  /** 閉じたタブを再度開く（Ctrl+Shift+T） */
  reopenClosedTab: () => void;

  /** タブをアクティブにする */
  setActiveTab: (groupId: string, tabId: string) => void;

  /** タブを移動する（ドラッグ＆ドロップ用） */
  moveTab: (fromGroupId: string, tabId: string, toGroupId: string, toIndex: number) => void;

  /** 画面を分割する */
  splitTab: (groupId: string, tabId: string, direction: SplitDirection) => void;

  /** 絶対パスをクリップボードにコピー */
  copyAbsolutePath: (tabId: string) => Promise<void>;

  /** 相対パスをクリップボードにコピー（OS形式） */
  copyRelativePath: (tabId: string) => Promise<void>;

  /** 相対パスをクリップボードにコピー（Posix形式） */
  copyRelativePathPosix: (tabId: string) => Promise<void>;

  /** OSエクスプローラーで表示 */
  revealInOsExplorer: (tabId: string) => Promise<void>;

  /** サイドバーのエクスプローラーで表示 */
  revealInSidebarExplorer: (tabId: string) => void;

  /** カーソル位置を更新する */
  updateCursorLine: (tabId: string, line: number) => void;

  /** スクロール位置を更新する */
  updateScrollTop: (tabId: string, scrollTop: number) => void;
}
```

**内部管理:**

- `closedTabsStack: Tab[]` — 閉じたタブのスタック（最大20件）。`reopenClosedTab()` で利用
- `tabOrder: string[]` — Ctrl+Tab時の表示順（最近アクティブだったタブ順）

## 3. SearchStore

**ファイル:** `src/stores/search.ts`

```typescript
interface SearchState {
  // ----- State -----
  query: string;
  options: SearchOptions;
  result: SearchResult | null;
  isSearching: boolean;
  history: HistoryEntry[];
  /** 結果グループの折りたたみ状態（filePath → boolean） */
  collapsedGroups: Set<string>;

  // ----- Actions -----

  /** 検索クエリを設定する（デバウンス300msで自動検索） */
  setQuery: (query: string) => void;

  /** 検索オプションを更新する */
  setOptions: (options: Partial<SearchOptions>) => void;

  /** 検索を実行する
   *  処理:
   *  1. isSearching = true
   *  2. invoke("search_fulltext", { query, options })
   *  3. result に結果をセット
   *  4. isSearching = false
   */
  executeSearch: () => Promise<void>;

  /** 検索履歴を読み込む */
  loadHistory: () => Promise<void>;

  /** 結果グループの折りたたみをトグルする */
  toggleGroupCollapse: (filePath: string) => void;

  /** 結果をクリアする */
  clearResults: () => void;
}
```

## 5. HighlightStore

**ファイル:** `src/stores/highlight.ts`

フロントエンドのみで完結（バックエンド通信なし）。セッション単位。

```typescript
interface HighlightState {
  // ----- State -----
  highlights: HighlightWord[];
  nextColorIndex: number;           // 0-19 ローテーション

  // ----- Actions -----

  /** ハイライトを追加する
   *  - 同一テキスト（ignoreCase考慮）が既に存在する場合は追加しない
   *  - nextColorIndex から色を割り当て、nextColorIndex = (nextColorIndex + 1) % 20
   */
  addHighlight: (text: string, ignoreCase?: boolean) => void;

  /** ハイライトを削除する */
  removeHighlight: (id: string) => void;

  /** 全ハイライトを削除する */
  removeAll: () => void;

  /** 次のマッチ位置を返す
   *  - 現在のカーソル位置から前方（下方向）に検索
   *  - ファイル末尾に到達したら先頭にラップ
   */
  navigateNext: (id: string, filePath: string, currentLine: number) => MatchPosition | null;

  /** 前のマッチ位置を返す
   *  - 現在のカーソル位置から後方（上方向）に検索
   *  - ファイル先頭に到達したら末尾にラップ
   */
  navigatePrev: (id: string, filePath: string, currentLine: number) => MatchPosition | null;
}
```

## 7. NotificationStore

**ファイル:** `src/stores/notification.ts`

```typescript
interface NotificationState {
  // ----- State -----
  notifications: Notification[];

  // ----- Actions -----

  /** 通知を追加する
   *  - id, createdAt を自動付与
   *  - autoCloseMs が設定されていれば setTimeout で自動削除
   *  戻り値: 生成された通知ID
   */
  addNotification: (n: Omit<Notification, 'id' | 'createdAt'>) => string;

  /** 進捗通知を更新する */
  updateProgress: (id: string, progress: { current: number; total: number }) => void;

  /** 通知を削除する */
  removeNotification: (id: string) => void;
}
```
