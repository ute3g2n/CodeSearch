# 03. フロントエンド型定義

`src/ipc/types.ts` に定義する、IPC通信で使用する共有型と、フロントエンド固有の型を定義する。

---

## 1. IPC共有型 (src/ipc/types.ts)

```typescript
// ===== ワークスペース =====

export interface Workspace {
  id: string;
  path: string;
  name: string;
  lastOpenedAt: string; // ISO 8601
}

export interface WorkspaceInfo {
  workspace: Workspace;
  indexStatus: 'building' | 'ready' | 'empty';
  fileCount: number;
  hasIndexWriteLock: boolean;
}

// ===== ファイル =====

export interface FileNode {
  name: string;
  path: string;
  isDir: boolean;
  children: FileNode[] | null;
  extension: string | null;
  size: number;
}

export interface FileContent {
  path: string;
  content: string;
  encoding: string;
  lineCount: number;
  size: number;
}

export interface FileMatch {
  name: string;
  relativePath: string;
  absolutePath: string;
  score: number;
  matchedIndices: number[];
}

// ===== 検索 =====

export interface SearchOptions {
  caseSensitive: boolean;
  wholeWord: boolean;
  isRegex: boolean;
  includeGlob: string | null;
  excludeGlob: string | null;
  maxResults: number | null;
}

export interface SearchResult {
  groups: SearchResultGroup[];
  totalMatches: number;
  elapsedMs: number;
}

export interface SearchResultGroup {
  filePath: string;
  relativePath: string;
  matches: SearchMatch[];
}

export interface SearchMatch {
  lineNumber: number;
  lineContent: string;
  matchRanges: [number, number][];
}

export interface HistoryEntry {
  id: number;
  workspaceId: string;
  query: string;
  isRegex: boolean;
  caseSensitive: boolean;
  wholeWord: boolean;
  includeGlob: string | null;
  excludeGlob: string | null;
  resultCount: number | null;
  searchedAt: string;
}

// ===== インデックス =====

export interface IndexStatus {
  state: 'idle' | 'building' | 'ready' | 'error';
  documentCount: number;
  lastBuiltAt: string | null;
  errorMessage: string | null;
}

// ===== ブックマーク =====

export interface Bookmark {
  id: number;
  workspaceId: string;
  filePath: string;
  lineNumber: number;
  colorIndex: number; // 0-14
  previewText: string | null;
  createdAt: string;
}

// ===== 設定 =====

export interface AppConfig {
  editorFontFamily: string;
  editorFontSize: number;
  uiFontFamily: string;
  uiFontSize: number;
  minimapEnabled: boolean;
  language: 'ja' | 'en';
  excludePatterns: string[];
  lastWorkspaceId: string | null;
}

// ===== イベントペイロード =====

export interface IndexProgressPayload {
  indexed: number;
  total: number;
  currentFile: string;
}

export interface IndexReadyPayload {
  workspaceId: string;
  documentCount: number;
}

export interface IndexErrorPayload {
  message: string;
}

export interface IndexUpdatedPayload {
  affectedPaths: string[];
}

export interface FsChangedPayload {
  changes: FsChangeEntry[];
}

export interface FsChangeEntry {
  path: string;
  kind: 'created' | 'modified' | 'deleted';
}

export interface ConfigChangedPayload {
  config: AppConfig;
}

export interface WatcherErrorPayload {
  message: string;
}

// ===== エラー =====

export interface CommandError {
  code:
    | 'FILE_NOT_FOUND'
    | 'ENCODING_ERROR'
    | 'INDEX_ERROR'
    | 'INDEX_LOCK_UNAVAILABLE'
    | 'DATABASE_ERROR'
    | 'WATCHER_ERROR'
    | 'IO_ERROR'
    | 'INVALID_ARGUMENT'
    | 'INTERNAL_ERROR';
  message: string;
}
```

## 2. フロントエンド固有型

### 2.1 エディタ関連 (src/stores/editor.ts)

```typescript
/** タブの種類 */
export type TabKind = 'file' | 'search-editor' | 'plain-text' | 'welcome';

/** タブの共通情報 */
export interface Tab {
  id: string;                   // UUID v4
  kind: TabKind;
  title: string;                // タブに表示する名前
  /** file タブの場合のみ */
  filePath?: string;
  /** search-editor タブの場合の検索クエリ */
  searchQuery?: string;
  /** plain-text タブの場合のテキスト内容 */
  plainText?: string;
  /** スクロール位置の復元用 */
  scrollTop?: number;
  /** カーソル行（1-based） */
  cursorLine?: number;
}

/** エディタグループ（画面分割の1区画） */
export interface EditorGroup {
  id: string;                   // UUID v4
  tabs: Tab[];
  activeTabId: string | null;
}

/** 分割方向 */
export type SplitDirection = 'left' | 'right' | 'up' | 'down';
```

### 2.3 通知関連 (src/stores/notification.ts)

```typescript
export type NotificationType = 'info' | 'progress' | 'error';

export interface NotificationAction {
  label: string;
  onClick: () => void;
}

export interface Notification {
  id: string;                          // UUID v4
  type: NotificationType;
  message: string;
  progress?: {
    current: number;
    total: number;
  };
  actions?: NotificationAction[];
  /** 自動消去までのミリ秒（undefinedの場合は手動消去のみ） */
  autoCloseMs?: number;
  createdAt: number;                   // Date.now()
}
```
